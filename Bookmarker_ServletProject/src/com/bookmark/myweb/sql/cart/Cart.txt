# CREATE TABLE
-- 장바구니 생성
CREATE TABLE cart (
  cart_id NUMBER PRIMARY KEY,
  user_id NUMBER NOT NULL, 
  book_id NUMBER NOT NULL,  
  added_date DATE DEFAULT SYSDATE
);

-- cart_id 자동 생성을 위한 시퀀스 생성
CREATE SEQUENCE cart_seq START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;
