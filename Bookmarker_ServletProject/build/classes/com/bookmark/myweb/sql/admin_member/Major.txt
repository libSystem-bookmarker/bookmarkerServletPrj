# CREATE TABLE
-- 학부/학과 계층형 테이블
CREATE TABLE ACADEMIC_UNIT (
    UNIT_ID NUMBER PRIMARY KEY,
    UNIT_NAME VARCHAR2(100) NOT NULL,
    UNIT_TYPE VARCHAR2(20) CHECK (UNIT_TYPE IN ('FACULTY', 'DEPARTMENT')) NOT NULL,
    PARENT_ID NUMBER,
    CONSTRAINT FK_UNIT_PARENT FOREIGN KEY (PARENT_ID)
        REFERENCES ACADEMIC_UNIT(UNIT_ID)
);

-- MAJOR ID SEQUENCE -> PK (
CREATE SEQUENCE ACADEMIC_UNIT_SEQ
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;
